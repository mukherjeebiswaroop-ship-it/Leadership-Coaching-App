    const userMsg = { id: Date.now(), role: "user", text };

      const apiMessages = history.map((m) => ({
        role: m.role,
        content: m.text,

// ─── PERSONALITY ASSESSMENT ──────────────────────────────────────────────────

const LIKERT_LABELS = ["Strongly\nDisagree", "Disagree", "Neutral", "Agree", "Strongly\nAgree"];
const QUESTIONS_PER_PAGE = 5;
const TOTAL_PAGES = Math.ceil(HOGAN_QUESTIONS.length / QUESTIONS_PER_PAGE);

function computeHoganScores(answers) {
  const result = {};
  HOGAN_SCALES.forEach((scale) => {
    const qs = HOGAN_QUESTIONS.filter((q) => q.scale === scale.id);
    const sum = qs.reduce((acc, q) => {
      const raw = answers[q.id] ?? 3;
      return acc + (q.reverse ? 6 - raw : raw);
    }, 0);
    result[scale.id] = Math.round(((sum / qs.length - 1) / 4) * 100);
  });
  return result;
}

function getInterpBand(score) {
  if (score >= 80) return "veryHigh";
  if (score >= 65) return "high";
  if (score >= 45) return "average";
  if (score >= 30) return "low";
  return "veryLow";
}

function getInterpLabel(score) {
  const map = { veryLow: "Very Low", low: "Low", average: "Average", high: "High", veryHigh: "Very High" };
  return map[getInterpBand(score)];
}

            ))

      </div>

      {/* Scale profiles */}
      <div style={{ ...styles.card, marginBottom: 24 }}>

        </div>
        {HOGAN_SCALES.map((scale, si) => {
          const score = scores[scale.id] ?? 0;
          const band = getInterpBand(score);
          const label = getInterpLabel(score);
          return (
            <div key={scale.id} style={{ marginBottom: 20, paddingBottom: 20, borderBottom: si < HOGAN_SCALES.length - 1 ? `1px solid ${COLORS.border}` : "none" }}>
              <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", marginBottom: 6 }}>
                <div style={{ display: "flex", gap: 10, alignItems: "center" }}>
                  <span style={{ fontSize: 20 }}>{scale.icon}</span>
                  <div>
                    <div style={{ fontSize: 14, fontWeight: 600, color: scale.color }}>{scale.name}</div>
                    <div style={{ fontSize: 11, color: COLORS.muted }}>{scale.description}</div>
                  </div>
                </div>
                <div style={{ textAlign: "right", flexShrink: 0, marginLeft: 16 }}>
                  <div style={{ fontSize: 24, fontWeight: 700, color: scale.color, lineHeight: 1 }}>{score}</div>
                  <div style={{ ...styles.tag(scale.color), fontSize: 10, marginTop: 4 }}>{label}</div>
                </div>
              </div>
              <div style={styles.progressBar(score, scale.color)}>
                <div style={styles.progressFill(score, scale.color)} />
              </div>
              <div style={{ display: "flex", justifyContent: "space-between", fontSize: 9, color: COLORS.muted, margin: "4px 0 8px" }}>
                {["Very Low", "Low", "Average", "High", "Very High"].map((l) => <span key={l}>{l}</span>)}
              </div>
              <div style={{ fontSize: 13, lineHeight: 1.7, color: COLORS.text, fontStyle: "italic", background: COLORS.surface, padding: "10px 14px", borderRadius: 8, borderLeft: `3px solid ${scale.color}` }}>
                {scale.interpretations[band]}
              </div>
            </div>
          );
        })}
      </div>

      {/* Strengths & Development edges */}
      <div style={{ ...styles.grid2, marginBottom: 24 }}>
        <div style={{ ...styles.card, border: `1px solid ${COLORS.green}44` }}>
          <div style={{ fontSize: 12, color: COLORS.green, fontWeight: 600, marginBottom: 16, letterSpacing: "0.08em" }}>SIGNATURE STRENGTHS</div>
          {strengths.map((scale, i) => (
            <div key={scale.id} style={{ display: "flex", gap: 12, paddingBottom: 14, marginBottom: i < strengths.length - 1 ? 14 : 0, borderBottom: i < strengths.length - 1 ? `1px solid ${COLORS.border}` : "none" }}>
              <span style={{ fontSize: 22 }}>{scale.icon}</span>
              <div>
                <div style={{ fontSize: 13, fontWeight: 600, color: scale.color, marginBottom: 4 }}>
                  {scale.name} <span style={{ fontWeight: 400, color: COLORS.muted }}>· {scores[scale.id]}/100</span>
                </div>
                <div style={{ fontSize: 12, color: COLORS.muted, lineHeight: 1.6 }}>
                  {scale.interpretations[getInterpBand(scores[scale.id])]}
                </div>
              </div>
            </div>
          ))}
        </div>

        <div style={{ ...styles.card, border: `1px solid ${COLORS.gold}44` }}>
          <div style={{ fontSize: 12, color: COLORS.gold, fontWeight: 600, marginBottom: 16, letterSpacing: "0.08em" }}>DEVELOPMENT OPPORTUNITIES</div>
          {edges.map((scale, i) => (

function PersonalityAssessmentPage() {
  const [phase, setPhase] = useState("intro");
  const [answers, setAnswers] = useState({});
  const [page, setPage] = useState(0);
  const [scores, setScores] = useState({});
  const [aiNarrative, setAiNarrative] = useState("");
  const topRef = useRef(null);

  const pageQuestions = HOGAN_QUESTIONS.slice(page * QUESTIONS_PER_PAGE, (page + 1) * QUESTIONS_PER_PAGE);
  const answered = Object.keys(answers).length;
  const progress = (answered / HOGAN_QUESTIONS.length) * 100;
  const pageComplete = pageQuestions.every((q) => answers[q.id] !== undefined);
  const isLastPage = page === TOTAL_PAGES - 1;

  function scrollTop() { topRef.current?.scrollIntoView({ behavior: "smooth" }); }

  function nextPage() { if (pageComplete) { setPage((p) => p + 1); scrollTop(); } }
  function prevPage() { setPage((p) => Math.max(0, p - 1)); scrollTop(); }

  async function submitAssessment() {
    const computed = computeHoganScores(answers);
    setScores(computed);
    setPhase("generating");

    const summary = HOGAN_SCALES.map((s) => `- ${s.name}: ${computed[s.id]}/100 (${getInterpLabel(computed[s.id])})`).join("\n");

    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1500,
          system: `You are an expert organizational psychologist specializing in the Hogan Personality Inventory (HPI). Generate honest, incisive, and constructive personality reports grounded strictly in the score data provided.`,
          messages: [{
            role: "user",
            content: `A leader has completed a 50-question Hogan-based personality assessment. Write a personalized narrative report in exactly 4 paragraphs — no headers, no bullets, pure prose separated by double newlines, written in second person ("You…"):

Paragraph 1 — LEADERSHIP IDENTITY (2-3 sentences): Describe their core leadership character based on their highest and lowest scores. Be precise and avoid generic language.

Paragraph 2 — SIGNATURE STRENGTHS (3-4 sentences): Name the specific strengths their profile predicts, grounded in the scale scores. Be specific about how these show up in leadership contexts.

Paragraph 3 — DEVELOPMENT EDGES (3-4 sentences): Identify the growth areas implied by their lowest scores, framed as genuine opportunities rather than deficits. Be constructive but honest.

Paragraph 4 — STRATEGIC PRIORITY (2-3 sentences): Give one clear, high-impact development recommendation that is directly tied to the profile. Make it actionable.

Hogan Profile:
${summary}`,
          }],
        }),
      });
      const data = await res.json();
      setAiNarrative(data.content?.[0]?.text ?? "");
    } catch {
      setAiNarrative("Your personality profile reveals a distinctive leadership identity. Connect with your coach to explore the full implications of your Hogan results.");
    }
    setPhase("report");
  }

  function retake() { setPhase("intro"); setAnswers({}); setPage(0); setScores({}); setAiNarrative(""); }

  // ── INTRO ──
  if (phase === "intro") {
    return (
      <div>
        <div style={styles.pageTitle}>Hogan Personality Assessment</div>
        <div style={styles.pageSubtitle}>A validated framework for understanding your leadership identity</div>

        <div style={{ display: "grid", gridTemplateColumns: "1.5fr 1fr", gap: 24, marginBottom: 24 }}>
          <div style={styles.card}>
            <div style={{ fontSize: 12, color: COLORS.accent, fontWeight: 600, marginBottom: 12, letterSpacing: "0.08em" }}>ABOUT THIS ASSESSMENT</div>
            <p style={{ fontSize: 14, lineHeight: 1.8, marginBottom: 16, marginTop: 0 }}>
              This 50-question assessment is modeled on the Hogan Personality Inventory (HPI) — one of the most widely validated personality instruments used in executive selection and leadership development globally. It measures how you typically present yourself in professional contexts across seven personality scales.
            </p>
            <p style={{ fontSize: 14, lineHeight: 1.8, color: COLORS.muted, fontStyle: "italic", marginBottom: 0, marginTop: 0 }}>
              There are no right or wrong answers. Respond based on how you actually are, not how you think you should be. Authenticity produces the most useful results.
            </p>
          </div>
          <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>
            {[
              { label: "Questions", value: "50", icon: "📝" },
              { label: "Estimated time", value: "10–15 min", icon: "⏱️" },
              { label: "Personality scales", value: "7", icon: "📊" },
            ].map((s) => (
              <div key={s.label} style={{ ...styles.card, display: "flex", alignItems: "center", gap: 16, padding: 16 }}>
                <span style={{ fontSize: 22 }}>{s.icon}</span>
                <div>
                  <div style={{ fontSize: 20, fontWeight: 700, color: COLORS.accent }}>{s.value}</div>
                  <div style={{ fontSize: 12, color: COLORS.muted }}>{s.label}</div>
                </div>
              </div>
            ))}
          </div>
        </div>

        <div style={{ ...styles.card, marginBottom: 24 }}>
          <div style={{ fontSize: 12, color: COLORS.accent, fontWeight: 600, marginBottom: 16, letterSpacing: "0.08em" }}>THE 7 HOGAN SCALES MEASURED</div>
          <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 0 }}>
            {HOGAN_SCALES.map((scale, i) => (
              <div key={scale.id} style={{ display: "flex", gap: 12, alignItems: "flex-start", padding: "12px 0", borderBottom: i < HOGAN_SCALES.length - 2 ? `1px solid ${COLORS.border}` : "none", paddingRight: i % 2 === 0 ? 24 : 0, paddingLeft: i % 2 === 1 ? 24 : 0, borderRight: i % 2 === 0 ? `1px solid ${COLORS.border}` : "none" }}>
                <span style={{ fontSize: 22 }}>{scale.icon}</span>
                <div>
                  <div style={{ fontSize: 13, fontWeight: 600, color: scale.color }}>{scale.name}</div>
                  <div style={{ fontSize: 12, color: COLORS.muted, lineHeight: 1.5 }}>{scale.description}</div>
                </div>
              </div>
            ))}
          </div>
        </div>

        <div style={{ display: "flex", alignItems: "center", gap: 20 }}>
          <button style={{ ...styles.btn("primary"), padding: "14px 40px", fontSize: 15 }} onClick={() => setPhase("questions")}>
            Begin Assessment →
          </button>
          <div style={{ fontSize: 12, color: COLORS.muted, fontStyle: "italic" }}>
            Rate each statement 1 (Strongly Disagree) → 5 (Strongly Agree)
          </div>
        </div>
      </div>
    );
  }

  // ── GENERATING ──
  if (phase === "generating") {
    return (
      <div style={{ display: "flex", flexDirection: "column", alignItems: "center", justifyContent: "center", minHeight: "60vh", textAlign: "center" }}>
        <div style={{ fontSize: 52, marginBottom: 20 }}>🧠</div>
        <div style={{ fontSize: 22, fontWeight: 600, marginBottom: 10 }}>Analyzing Your Profile</div>
        <div style={{ fontSize: 14, color: COLORS.muted, fontStyle: "italic", marginBottom: 32 }}>
          Scoring 7 personality scales and generating your personalized Hogan report…
        </div>
        <Spinner />
      </div>
    );
  }

  // ── REPORT ──
  if (phase === "report") {
    return <PersonalityReport scores={scores} aiNarrative={aiNarrative} onRetake={retake} />;
  }

  // ── QUESTIONS ──
  return (
    <div>
      <div ref={topRef} style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginBottom: 6 }}>
        <div style={styles.pageTitle}>Hogan Personality Assessment</div>
        <div style={{ fontSize: 13, color: COLORS.muted }}>Page {page + 1} of {TOTAL_PAGES} · {answered}/50</div>
      </div>

      <div style={{ marginBottom: 24 }}>
        <div style={styles.progressBar(progress)}>
          <div style={styles.progressFill(progress, COLORS.accent)} />
        </div>
        <div style={{ fontSize: 11, color: COLORS.muted, marginTop: 4 }}>{Math.round(progress)}% complete</div>
      </div>

      {pageQuestions.map((q) => {
        const scale = HOGAN_SCALES.find((s) => s.id === q.scale);
        return (
          <div key={q.id} style={{ ...styles.assessmentQ, marginBottom: 16 }}>
            <div style={{ display: "flex", alignItems: "center", gap: 8, marginBottom: 10 }}>
              <span style={{ fontSize: 14 }}>{scale.icon}</span>
              <span style={styles.tag(scale.color)}>{scale.name}</span>
              <span style={{ fontSize: 11, color: COLORS.muted }}>Q{q.id}</span>
            </div>
            <div style={{ fontSize: 15, fontWeight: 600, lineHeight: 1.6, marginBottom: 16 }}>{q.text}</div>
            <div style={{ display: "grid", gridTemplateColumns: "repeat(5, 1fr)", gap: 8 }}>
              {LIKERT_LABELS.map((label, idx) => {
                const val = idx + 1;
                const selected = answers[q.id] === val;
                return (
                  <button
                    key={val}
                    onClick={() => setAnswers({ ...answers, [q.id]: val })}
                    style={{
                      padding: "10px 4px",
                      borderRadius: 8,
                      border: `1px solid ${selected ? scale.color : COLORS.border}`,
                      background: selected ? scale.color + "22" : "transparent",
                      color: selected ? scale.color : COLORS.muted,
                      cursor: "pointer",
                      fontSize: 11,
                      fontFamily: "'Georgia', serif",
                      textAlign: "center",
                      lineHeight: 1.4,
                      display: "flex",
                      flexDirection: "column",
                      alignItems: "center",
                      gap: 4,
                      transition: "all 0.15s",
                    }}
                  >
                    <span style={{ fontSize: 17, fontWeight: 700 }}>{val}</span>
                    <span style={{ whiteSpace: "pre-line" }}>{label}</span>
                  </button>
                );
              })}
            </div>
          </div>
        );
      })}

      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginTop: 8 }}>
        <button
          style={{ ...styles.btn("outline"), opacity: page === 0 ? 0.3 : 1 }}
          onClick={prevPage}
          disabled={page === 0}
        >
          ← Previous
        </button>

        {!isLastPage ? (
          <button
            style={{ ...styles.btn("primary"), opacity: pageComplete ? 1 : 0.4 }}
            onClick={nextPage}
            disabled={!pageComplete}
          >
            Next →
          </button>
        ) : (
          <button
            style={{ ...styles.btn("gold"), padding: "12px 28px", opacity: answered === 50 ? 1 : 0.4 }}
            onClick={() => { if (answered === 50) submitAssessment(); }}
            disabled={answered < 50}
          >
            Generate My Report →
          </button>
        )}
      </div>
    </div>
  );
}
